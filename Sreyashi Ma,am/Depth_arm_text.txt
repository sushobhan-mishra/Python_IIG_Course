

"""
====================================================================
DEPTH vs ARM PLOT
====================================================================

Author      : Your Name
Version     : 1.0

Description
-----------
Reads ARM (Anhysteretic Remanent Magnetization) data and depth data
from an Excel file and creates a publication-quality geological plot.

Input File
----------
Depth_arm.xlsx

Required Columns
----------------
ARM
Top depth [m CSF-A]

Output File
-----------
Depth_vs_ARM_400dpi.jpg

Required Packages
-----------------
pip install pandas matplotlib openpyxl

Run
---
python depth_vs_ARM.py

====================================================================
"""

# ------------------------------------------------------------------
# IMPORT LIBRARIES
# ------------------------------------------------------------------

import pandas as pd
import matplotlib.pyplot as plt

# ------------------------------------------------------------------
# READ EXCEL FILE
# ------------------------------------------------------------------

excel_file = "Depth_arm.xlsx"

df = pd.read_excel(excel_file)

# ------------------------------------------------------------------
# EXTRACT DATA
# ------------------------------------------------------------------

arm = df["ARM"]
depth = df["Top depth\n[m CSF-A]"]

# ------------------------------------------------------------------
# CREATE FIGURE
# ------------------------------------------------------------------

fig, ax = plt.subplots(figsize=(7, 10))

# ------------------------------------------------------------------
# PLOT DATA
# ------------------------------------------------------------------

ax.plot(
    arm,
    depth,
    linestyle="-",
    linewidth=1.2,
    marker="o",
    markersize=3
)

# ------------------------------------------------------------------
# GEOLOGICAL CONVENTION
# DEPTH INCREASES DOWNWARD
# ------------------------------------------------------------------

ax.invert_yaxis()

# ------------------------------------------------------------------
# AXIS LABELS
# ------------------------------------------------------------------

ax.set_xlabel(
    "ARM",
    fontsize=12
)

ax.set_ylabel(
    "Depth (m CSF-A)",
    fontsize=12
)

# ------------------------------------------------------------------
# TITLE
# ------------------------------------------------------------------

ax.set_title(
    "Depth vs ARM",
    fontsize=14,
    fontweight="bold"
)

# ------------------------------------------------------------------
# GRID
# ------------------------------------------------------------------

ax.grid(
    True,
    linestyle="--",
    linewidth=0.5,
    alpha=0.6
)

# ------------------------------------------------------------------
# IMPROVE LAYOUT
# ------------------------------------------------------------------

plt.tight_layout()

# ------------------------------------------------------------------
# SAVE FIGURE
# ------------------------------------------------------------------

output_file = "Depth_vs_ARM_400dpi.jpg"

plt.savefig(
    output_file,
    dpi=400,
    format="jpeg",
    bbox_inches="tight",
    facecolor="white"
)

print(f"Figure saved as: {output_file}")

# ------------------------------------------------------------------
# DISPLAY PLOT
# ------------------------------------------------------------------

plt.show()
